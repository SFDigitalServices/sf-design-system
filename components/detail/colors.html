{% for group in groups %}
  <h2>{{ group.title }}</h2>
  <ul class="sf-colors">
    {% for color in group.colors %}
      <li class="sf-color">
        <span class="sf-swatch" style="background-color: {{ colors[color] }};"></span>
        <span class="sf-label">
          ${{ color }}
          <br>
          <code>{{ colors[color] }}</code>
        </span>
      </li>
    {% endfor %}
  </ul>
{% endfor %}

<h2>Uncategorized</h2>
<ul class="sf-colors">
  {% for name, color in colors %}
    {% set grouped = false %}
    {% for group in groups %}
      {% if group.colors.indexOf(name) > -1 %}
        {% set grouped = true %}
      {% endif %}
    {% endfor %}
    {% if not grouped %}
      <li class="sf-color">
        <span class="sf-swatch" style="background-color: {{ color }};"></span>
        <span class="sf-label">
          ${{ name }}
          <br>
          <code>{{ color }}</code>
        </span>
      </li>
    {% endif %}
  {% endfor %}
</div>
